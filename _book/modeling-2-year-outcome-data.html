  cs_model |> 
  dplyr::select(
    Parameter,
    Coefficient,
    CI_low,
    CI_high,
    Group,
    Fit
    ) |> 
  dplyr::mutate(
    across(c(Coefficient, CI_low, CI_high),
          ~ round(.x, digits = 1)),
    across(c(Fit),
          ~ round(.x, digits = 2))
    ) |> 
  dplyr::mutate(
    Parameter =
      dplyr::case_when(
        is.na(Parameter) ~ Group,
        TRUE ~ as.character(Parameter)
        ),
    Coefficient =
      dplyr::case_when(
        is.na(Coefficient) ~ Fit,
        TRUE ~ Coefficient
      )
    ) |> 
  dplyr::select(
